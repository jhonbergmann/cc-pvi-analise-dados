import matplotlib.pyplot as plt
import pandas as pd

BrazilSuicidiosIdadeMasc = pd.read_csv('BrazilSuicidiosIdadeMasc.csv', header=0, index_col=0, parse_dates=True, squeeze=True, low_memory=False)
BrazilSuicidiosIdadeMasc.plot()

plt.title('Suicídios do Sexo Masculino no Brasil (1985-2015)')
plt.ylabel('Taxa de Suicídios')
plt.show()

import plotly.graph_objects as go 
import pandas as pd 

df = pd.read_csv('SouthAmericaSuicidios2003.csv') 

fig = go.Figure(data=go.Choropleth(locations=df['country'], z = df['year-2003'].astype(int), locationmode = 'country names',colorscale = ['orange','blue','red'], colorbar_title = "Índice Suicídios", )) 
fig.update_layout( title_text = 'Suicídios na America do Sul em (2003)', geo_scope='south america',) 
fig.show()

import plotly.express as px 
import pandas as pd 

df = pd.read_csv('Brazil2010MascFemi.csv')
idade = df['idade']
sexo = df['sexo']

quantidade = df['quantidade']
df = pd.DataFrame(dict(idade=idade,sexo=sexo,quantidade=quantidade)) 
df["all"] = "all" 
fig = px.treemap(df,path=[sexo,idade],values=quantidade) 
fig.update_layout( title_text = 'Suicídios no Brasil por Sexo e Idade (2010)',)
fig.show()
